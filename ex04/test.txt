#include "Zombie.hpp"

Zombie::Zombie( void ) {
	this->setName("nameless");
	std::cout << "Zombie " << this->name << " has been constructed" <<std::endl;
}

	std::cout << "Zombie " << this->name << " has been destroyed" <<std::endl;
}

}
